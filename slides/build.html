  • Command line interface written in Node.js
  • Open source under MIT Licence hosted on GitHub
  • Sponsored by Bocoup
  • Over 2700 plugins!

Grunt : Popular plugins

  • uglify: Minify files with Uglify.JS
  • jshint: Validate files with JSHint
  • watch: Run predefined tasks whenever a watched file changes.
  • coffee: Compile CoffeeScript files to Javascript
  • connect: Start connect (node.js) web server
  • less: Compile LESS files to CSS
  • imagemin: Minify images
  • and many more...

Configuration: Gruntfile.(js|cofee)


module.exports = function(grunt) {
  // Project configuration.
  grunt.initConfig({
    pkg: grunt.file.readJSON('package.json'),
    uglify: {
      options: {
        banner: '/*! <%= pkg.name %> <%= grunt.template.today("yyyy-mm-dd") %> */\n'
      },
      build: {
        src: 'src/<%= pkg.name %>.js',
        dest: 'build/<%= pkg.name %>.min.js'
      }
    }
  });

  // Load the plugin that provides the "uglify" task.
  grunt.loadNpmTasks('grunt-contrib-uglify');
  // Default task(s).
  grunt.registerTask('default', ['uglify']);
};
